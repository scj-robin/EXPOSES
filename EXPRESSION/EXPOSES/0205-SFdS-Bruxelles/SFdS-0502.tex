\subsection{1.3 -- Étude du transcriptome}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\paragraph{Idée~:}
Pour connaître la fonction des gène il faut savoir \textred{dans
  quelles conditions ou dans quels tissus} \textred{ils s'expriment}

\paragraph{Principe~:}
On mesure les niveaux d'expression de \textred{tous les gènes} au
travers le transcriptome (= \textgreen{ensemble des transcrits})

\section{3 -- QUESTION BIOLOGIQUES}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\paragraph{Profils d'expression~:}
Déterminer des groupes de gènes ayant des \textred{profils
  similaires} dans un ensemble de conditions

\paragraph{Gènes réagissant à un changement de condition~:}
Détecter des gènes ``\textred{différentiellement} exprimés''

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\newpage
\paragraph{Caractérisation de tumeurs~:}\\
$\bullet$ \textred{Reconnaître} une tumeur à partir des niveaux d'expression \\
$\bullet$ Fabriquer des \textred{mini-puces} pour des analyses en routine

\paragraph{Réseaux de régulation~:}
Déterminer les actions de régulation (activation ou inhibition) des
gènes \textred{les uns sur les autres}

\subsection{4.5 -- Planification expérimentale}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\paragraph{``Swap''~:} inversion des couleurs entre deux lames
$$
\mbox{\textred{carré latin}~:} \qquad
\begin{tabular}{c|cc}
  & Lame 1 & Lame 2 \\
  \hline
  Rouge & Cond. 1 & Cond. 2 \\
  Vert & Cond. 2 & Cond. 1 \\
\end{tabular}
$$

\paragraph{Utilisation d'un contrôle~:} 
$$
\begin{tabular}{cl}
  $\bullet$ & comparaison de toutes les conditions au contrôle \\
  & C1/T \quad C2/T \quad C3/T \\
  \\
  $\bullet$ & plan ``\textred{en boucle}'' \\
  & T/C1 \quad C1/C2 \quad C2/C3 \quad C3/T
\end{tabular}
$$

\newpage
\subsection{4.6 -- Analyse différentielle} 
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

${\bf H}_0$ = ``le gène n'est pas différentiellement exprimé''

\paragraph{Test usuels~:} Student, Wilcoxon, etc

\paragraph{Loi sous ${\bf H}_0$~:} on ne connaît pas les gènes ``sous ${\bf H}_0$'' \\
$\rightarrow$ utilisation de gènes témoins \\
$\rightarrow$ bootstrap
