\subsection{Anova classique}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{eqnarray*}
Z_{gtr} &  = & \mbox{constante générale} \\
 & + & \mbox{correction pour le gène } g \\
 & + & \mbox{correction pour la condition } t \\
 & + & \mbox{interaction gène $g$ $\times$ condition $t$} \\
 & + & \mbox{erreur aléatoire } (g, t, r)
\end{eqnarray*}

\paragraph{Idée~:}\\ 
 -- Le terme intéressant est le terme d'{\bf interaction} \\
 -- Toute l'erreur de mesure est prise en compte par le dernier
    terme\\
 -- Cette erreur est purement additive \\
 -- C'est sur cette erreur que portent toutes les hypothèses
  statistiques

\subsection{Anova sur les rangs}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\paragraph{Principe}
\begin{enumerate}
\item On classe les $G \times T \times R$ mesures $Z_{gtr}$ par ordre
  croissant et on leur associe leur rang dans ce classement:
  $$
  Z_{gtr} \rightarrow \mbox{Rang}(Z_{gtr})
  $$
\item On oublie les mesures originales $Z_{gtr}$ et on fait une anova
  sur les {Rang}$(Z_{gtr})$
\end{enumerate}

\paragraph{Intérêt}
\begin{itemize}
\item Plus d'hypothèse de normalité
\item Mais toujours une hypothèse de variance homogène (moins cruciale)
\end{itemize}

\subsection{Algorithme EM}\\
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\paragraph{Initialisation~:}\\
On choisit des valeurs initiales $(p_-, p_0, p_+)^0$, $(\mu_-,
\mu_+)^0$, $(\sigma_-, \sigma_0, \sigma_+)^0$\\
~\\
\paragraph{Étape E (estimation)~:}\\
On calcule la probabilité de chaque gène d'appartenir à chaque classe
avec les paramètres de l'étape précédente ($i$): $p_-^i(g)$,
$p_0^i(g)$, $p_+^i(g)$\\
~\\\paragraph{Étape M (maximisation)~:}\\
On ré-estime les paramètres $(p_-, p_0, p_+)^{i+1}$, $(\mu_-,
\mu_+)^{i+1}$, $(\sigma_-, \sigma_0, \sigma_+)^{i+1}$ avec les
probabilités $p^i(g)$:
$$
\mu_-^{i+1} = \frac{\sum_g p_-^i(g) D_g}{\sum_g p_-^i(g)}
$$
\paragraph{XEMgauss~:}
