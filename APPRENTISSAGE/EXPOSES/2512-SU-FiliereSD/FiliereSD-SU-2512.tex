\paragraph{Le cours de sciences des données de L1.}
\begin{itemize}
  \setlength{\itemsep}{0.75\baselineskip}
  \item ``Données'' absolument partout
  \item Nécessité d'une démarche scientifique pour les analyser

\end{itemize}
