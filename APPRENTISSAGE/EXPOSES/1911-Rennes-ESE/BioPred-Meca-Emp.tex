\subsection{Modèles empiriques vs mécanistes}
\frame{\frametitle{Outline} \tableofcontents[currentsubsection]}
%====================================================================
\frame{\frametitle{Modèles empiriques / mécanistes}

  \paragraph{Breiman (2001) :} '{\sl Statistical Modeling: The Two Cultures}'
  \begin{itemize}
   \item mécaniste : {\sl assumes that the data are generated by a given stochastic \textbf{data model}}
   \item empirique : {\sl uses \textbf{algorithmic models} and treats the data mechanism as unknown}
  \end{itemize}
  
  \bigskip \bigskip \pause
  \paragraph{Parti pris de l'apprentissage:}
  \begin{itemize}
   \item inutile de comprendre pour prédire
   \item seulement comprendre pour agir (au mieux) 
  \end{itemize}
  \ra Prédiction et optimisation \\
  \ra Pas d'analyse des mécanismes
  
  \bigskip \bigskip \pause
  \paragraph{En pratique :} approches hydrides

}

\frame{\frametitle{Approche hybride : sélection génomique}

  \paragraph{Prédiction d'un phénotype} à partir du génotype :  \\ ~ \\
  \begin{itemize}
  \item Modèle empirique de prédiction fondé sur une population contemporaine \\ ~
  \item Modèle génétique d'évolution de population (soumise à sélection) \\ ~
  \item Estimation de la qualité des prédictions dans les générations futures
  \end{itemize}
