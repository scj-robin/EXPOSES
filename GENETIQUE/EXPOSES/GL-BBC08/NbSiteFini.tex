\begin{frame}
  \frametitle{Modèle à nombre fini d'allèles}
  \begin{itemize}
  \item Les mutations sont vues comme des transitions entre $A$
    allèles possibles.
  \item Elles surviennent selon un processus Markovien homogène à
    temps continu $\{X_t\}_{t \geq 0}$.
  \end{itemize}
  \pause
  On note $r_{ab}$ le taux de transition de l'allèle $a$ vers l'allèle
  $b$ et $R = [r_{ab}]$ le générateur infinitésimal du
    processus. %($r_{aa} = -\sum_{b \neq a} r_{ab}$).
  \begin{itemize}
  \item La matrice de transition en une unité de temps est $\exp(R)$.
  \item Plus généralement~: 
    $$
    \pi_{ab}(s) := P(X_{t+s} = b | X_t = a) = [exp(s R)]_{ab}.
    $$
  \end{itemize}
  
  $\pi_{66}(t_1)$ ne signifie pas qu'il n'y a pas eu de mutation entre
  temps.
\end{frame}
