\frame{ \frametitle{Statistical model}

  \paragraph{Observed data.} The adundance vector in site $i$:
  $$
  y_i = [y_{i1} \dots y_{ij} \dots y_{im}]
  $$
  is seen as a realisation of a random vector
  $$
  Y = [Y_1 \dots Y_j \dots Y_m]
  $$
  (For the time being, forget about the covariates: same $x_i$ in all sites.)

  \bigskip \bigskip \pause
  \paragraph{Statistical model.} 
  $$
  Y \sim p \quad \text{= joint distribution: } p(Y_1, \dots, Y_m)
  $$

  \bigskip \bigskip \pause
  \paragraph{Where is the network (graph)?} \\ ~
  \begin{itemize}
    \setlength{\itemsep}{1\baselineskip}
    \item The graph $G$ has to be encoded in the distribution $p$ in some way 
    \item \emphase{Graphical models} establish a formal connexion between $p$ and $G$
  \end{itemize}

}

\frame{\frametitle{A reminder in probability}

\paragraph{Two random variables $A$ and $B$.}
\begin{align*}
  \text{Joint distribution:} & & 
  p(a, b) & = \Pr\{A=a, B=b\} \\ 
  \text{Marginal distribution:} & & 
  p(a) & = \Pr\{A=a\} = \sum_{b \in \Bcal} p(a, b) \\ 
  \text{Conditional distribution:} & & 
  p(b \mid a) & = \Pr\{B=b \mid A=a\} = \displaystyle{\frac{p(a, b)}{p(a)}}
\end{align*}

\bigskip \bigskip \pause
\paragraph{Factorisation.}
As a consequence:
\begin{align*}
p(a, b) 
& = p(a) \; p(b \mid a) & & \text{'$A$, then $B$ given $A$'} & & A \to B \\ 
\\
& = p(b) \; p(a \mid b) & & \text{'$B$, then $A$ given $B$'} & & B \to A
\end{align*}

\frame{\frametitle{Directed graphical models = Bayesian networks}

  \paragraph{Definition.} Let $D$ be a \emphase{directed acyclic graph} (DAG), the distribution $p$ is said to factorize in $D$ iff
  $$
  p(a_1, \dots a_m) = \prod_{j=1}^m p(a_j \mid a_{pa_D(j)})
  $$
  where $pa_D(i)$ stands for the set of parents of $i$ in $D$.

  \pause
  \begin{center}
  \begin{tabular}{cc}
    \begin{tabular}{c}
    $\footnotesize{\input{\figeco/DAG6nodes}}$
    \end{tabular}
    &
    \begin{tabular}{p{.7\textwidth}}
      \begin{eqnarray*}
        pa_D(A) = \emptyset, & & pa_D(D) = \{B, C\} \\
        \\
        p(a, \dots f) & = 
        & p(a) \; p(b \mid a) \; p(c \mid a) \\
        & & p(d \mid b, c) \; p(e \mid d) \\
        & & p(f \mid b, d)
      \end{eqnarray*}
      \\
      See \refer{SWA17} for an introduction in ecology
    \end{tabular}
  \end{tabular}
  \end{center}
  
}

  Consider $D =$
  $$
  \input{\figeco/LeftRightChainABC}
  $$
  $p(a, b, c)$ is faithful to $D$ iff
  $$
  p(a, b, c) = p(a) \; p(b \mid a) \; p(c \mid a)

  \begin{align*}
   p(a) \; p(b \mid a) \; p(c \mid b) 
   & = p(a \mid b) \; p(b \mid c) \; p(c) 
  \end{align*}
  so $p$ is also faithful to $D' =$
  $$
  \input{\figeco/RightLeftChainABC} 
  $$
  \pause and to $D'' =$
  $$
  \input{\figeco/AwayFromCenterABC}
  $$

  \paragraph{Conclusions.} 
  \begin{itemize}
   \item $p(a, b, c, \dots)$ is not enough to retrieve the edge orientations (see appendix)

  \end{itemize}

\frame{\frametitle{Gaussian graphical models (GGM)}

  \paragraph{Gaussian setting.} Suppose we observe an 'abundance' vector $Y = (Y_1 \dots Y_m)$ and assume that
  $$
  Y \sim \Ncal_m(\mu, \Sigma)
  $$
  \begin{itemize}
   \item $\mu = (m \times 1)$ vector of mean 'abundances'
   \item $\Sigma = (m \times m)$ covariance matrix:
   $$
   \sigma_{jj} = \sigma_j^2 = \Var(Y_j), \qquad
   \sigma_{jk} = \Cov(Y_j, Y_k), 
   $$
   \item Correlation:
   $$
   \rho_{jk} = \frac{\sigma_{jk}}{\sigma_j \; \sigma_k}
   $$
  \end{itemize}

  \bigskip \pause
  \paragraph{Property.} For species $j$ and $k$,
  $$
  \begin{array}{lcccccc}
    & & 
    \text{null correlation} & \Leftrightarrow & 
    \text{null covariance} & \Leftrightarrow & 
    \text{independence} \\
    ~ \\
    \text{i.e.} 
    & & \rho_{jk} = 0 
    & \Leftrightarrow &
    \sigma_{jk} = 0 
    & \Leftrightarrow &
    Y_j \independent Y_k    
  \end{array}
  $$
  but we care about emphase{conditional dependency}.

}

\frame{\frametitle{Network inference}

  \paragraph{Aim.} Given a series of iid 'abundance' vectors $Y_1, \dots Y_n$, find the graph $G$ that best fits the distribution of the $Y_i$'s.

  \bigskip \pause
  \paragraph{A huge search space:}
  $$
  \includegraphics[trim=50 50 0 75, height=.45\textheight, clip]{\fignet/NbGraphs2}
  $$
  Number of \textcolor{red}{directed acyclic graphs} (no close form), \textcolor{darkgreen}{undirected graphs} ($2^{m(m-1)/2}$), \textcolor{blue}{spanning trees} ($m^{m-2}$).
  $x$-axis: $m =$ number of species.
  
  \bigskip
  $\to$ Need for efficient algorithms / heuristics

}

\frame{\frametitle{Making $\Omega$ sparse}

  \paragraph{Graphical lasso \refer{FHT08}.} Penalized likelihood:
  $$
  \max_{\mu, \Omega} \quad \underset{\text{\footnotesize{log-likelihood}}}{\underbrace{\log p(Y; \mu, \Omega)}} - \underset{\text{\footnotesize{penalization}}}{\underbrace{\lambda \sum_{j < k} |\omega_{jk}|}}
  $$
  the $\ell_1$ penalty ('lasso') forces some $\omega_{jk}$ to be zero. \\ ~
  
  \begin{itemize}
    \setlength{\itemsep}{1\baselineskip}
    \item The higher $\lambda$, the sparser the network
    \item Turns out to be a convex problem, so fast solutions exists ({\tt huge} R package)
  \end{itemize}

  \bigskip \bigskip \pause
  \paragraph{Alternatives.}
  \begin{itemize}
   \item Regression viewpoint (see appendix) 
   \item Mixture of tree-shaped distributions: \refer{MeJ06,Kir07,JFS16,ScR17,MRA19}
   \item Other forms of penalties ($\ell_1$, $\ell_2$, combinations) induces other topologies \\
   Ex.: \refer{ACM09} induce SBM structured networks.
  \end{itemize}
}
