\subsection{Joint species distribution models}
\frame{\frametitle{Outline} \tableofcontents[currentsubsection]}
%====================================================================
\frame{\frametitle{Back to ecology}

  \paragraph{Specificities.} 
  \begin{itemize}
    \setlength{\itemsep}{1\baselineskip}
    \item Mostly count or presence/absence data, which do not fit a Gaussian distribution \item Environmental effects need to be accounted for
  \end{itemize}

  \bigskip \bigskip \pause
  \paragraph{Joint species distribution models (JSDM).}
  \begin{itemize}
    \setlength{\itemsep}{.75\baselineskip}
    \item $n$ sites, $m$ species
    \item $Y_i = [Y_{i1}, \dots Y_{im}]$ abundance vector observed in site $i$
    \item $x_i =$ vector of environmental covariates for site $i$ \textcolor{gray}{+ species traits $t_j$}
    \item JSDM: multivariate distribution
    $$
    p(Y_i \mid x_i)
    $$
    \item See \refer{WBO15} for a general introduction to JSDM or \refer{OvA20} for some R tools
  \end{itemize}

  \bigskip \bigskip 
  \paragraph{Remark.} Univariate of JSDM = SDM $\simeq$ Generalized linear models (glm / glmm).

}

%====================================================================
\frame{\frametitle{Network inference with JSDM}

  \paragraph{Problem.}
  \begin{itemize}
    \item Not many flexible {\sl multivariate} distributions for count or binary data do exist \refer{IYA17}
  \end{itemize}

  \bigskip \bigskip \pause
  \paragraph{A common trick:} latent variable models
  \begin{itemize}
    \setlength{\itemsep}{1\baselineskip}
    \item Most popular structure = latent GGM: \\ 
    Spiec-Easi \refer{KMM15}, gCODA \refer{FHZ17}, MiNT \refer{BML16}, PLNnetwork \refer{CMR18b}, tree-based PLN \refer{MRA19} 
    \item PLN = Poisson log-normal model (see next)
  \end{itemize}
  
  \bigskip \bigskip \pause
  \paragraph{But also:} 
  \begin{itemize}
    \item Copula-based approaches \refer{AVP19}
  \end{itemize}
  
}

\subsection{Poisson log-normal model}
\frame{\frametitle{Outline} \tableofcontents[currentsubsection]}
%==================================================================
\frame{\frametitle{Poisson log-normal model}

  \paragraph{Data:} $n$ independent sites (no spatial structure), $m$ species, 
  $$
  Y_{ij} = \text{ abundance of species $j$ in site $i$}
  $$
  
  \bigskip \pause
  \paragraph{Poisson log-normal (PLN) model \refer{AiH89,CMR21}:} 
  \begin{itemize}
   \item A Gaussian vector is associated with each site $i$
   $$
   Z_i \sim \Ncal_m(0, \Sigma)
   $$
   \item \pause Species abundances in site $i$ are independent given $Z_i$
   $$
   Y_{ij} \sim \Pcal(\exp(x_i^\top \beta_j + Z_{ij}))
   $$
  \end{itemize} \pause
  summarized as $\{Y_i\}_{1 \leq i \leq n}$ independent
  $$
  Y_i \sim PLN(x_i, \beta, \Sigma),
  $$
  $\to$ multivariate mixed generalized linear model (glmm). 

  \bigskip \pause
  \paragraph{Interpretation.}
  \begin{itemize}
%   \item PLN is a mixed model
  \item $x_i^\top \beta_j$ mean (log-)abundance of species $j$: \emphase{abiotic effects}
  \item $\Sigma = $ dependency structure (encoded in the latent layer): \emphase{biotic interactions}
  \end{itemize}
}

%==================================================================
\frame{\frametitle{Network inference}

  \paragraph{PLN network model.} Same model as PLN + sparsity assumption:
  \begin{align*}
  \{Y_i\} & \text{ independent,} & 
  Y_i & \sim PLN(x_i, \beta, \Sigma), &
  + & \emphase{\Omega = \Sigma^{-1} \text{ sparse}}
  \end{align*}
  
  \bigskip \bigskip \pause
  \paragraph{Inference algorithm.} Variational EM + lasso penalty \refer{CMR18a,CMR19}:
  $$
  \max_{\beta, \Omega} \; \widetilde{\log p}(Y; x, \beta, \Sigma) - \lambda \sum_{j < k} |\omega_{jk}|
  $$
  
  \bigskip
  \begin{itemize}
    \setlength{\itemsep}{1\baselineskip}
    \item Alternate convex problems \\
    $\to$ Fast solution ({\tt PLNmodels} R package)
    \item 'Atutomatic' choice of $\lambda$: BIC, EBIC, cross-validation, \dots
   \item Resampling (StARS \refer{LRW10}) can be used to assess robustness
  \end{itemize}
}
