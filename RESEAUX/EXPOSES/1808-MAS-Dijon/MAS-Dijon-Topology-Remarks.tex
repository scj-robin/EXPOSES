  \paragraph{Network models.}
  \begin{itemize}
   \item Many others: preferential attachment, geometric graphs, ...
   \item ERGM and SBM are the most popular for applications
  \end{itemize}

  \pause\bigskip\bigskip
  \paragraph{Around SBM.}
  \begin{itemize}
   \item Dynamic networks: $Y_{ij} \rightarrow Y_{ij}(t)$ \refer{MaM16,MRV17}, L\'ea Longepierre's talk
   \item Accounting for the sampling strategy \Refer{TBC17}
   \item Variational approximation: works well in practice but no general theoretical analysis
  \end{itemize}
  
  \pause\bigskip\bigskip
  \paragraph{Some more questions.}
  \begin{itemize}
   \item Goodness-of-fit: Is the observed graph a realization of given random graph? \refer{ArV14,VeA15,LRO17}
   \item Network motifs (= subgraphs = patterns): Are some motifs unexpectedly frequent in a given graph? \refer{MSI02,SCB18} \nocite{ORP15}
  \end{itemize}
  \ra Need to account for the estimation of the parameters of the null model
