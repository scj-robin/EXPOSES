\frame{\frametitle{Approximate Louis formula} 

  \paragraph{Louis formula.} \refer{Lou82}
  $$
  \partial^2_{\theta^2} \log p_\theta(Y) = \Esp\left(\partial^2_{\theta^2} \log p_\theta(Y, Z) \gv Y\right) + \Var\left(\partial_{\theta} \log p_\theta(Y, Z) \gv Y\right)
  $$
  \ra Compute Fisher information using by-products of EM
  
  \bigskip\bigskip
  \paragraph{VEM approximation.} Replace the conditional moments with the approximate moments 
  $$
  \partial^2_{\theta^2} \log p_\theta(Y) \simeq \Espt\left(\partial^2_{\theta^2} \log p_\theta(Y, Z) \gv Y\right) + \Vart\left(\partial_{\theta} \log p_\theta(Y, Z) \gv Y\right)
  $$

  \bigskip\bigskip
  \paragraph{SBM case.} The variational distribution is very concentrated, so $\Vart\left(g(Z)\right) \simeq 0$:
  $$
  \partial^2_{\theta^2} \log p_\theta(Y) \simeq \Espt\left(\partial^2_{\theta^2} \log p_\theta(Y, Z) \gv Y\right)
  $$
