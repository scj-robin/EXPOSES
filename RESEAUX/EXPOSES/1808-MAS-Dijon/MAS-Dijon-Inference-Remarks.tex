\frame{\frametitle{Network inference: Some remarks} 

  \paragraph{General case.}
  \begin{itemize}
   \item Statistical guaranties exists for some specific models (Gaussian)
   \item Selecting the network density (i.e. tuning $\lambda$) is a major practical issue
   \item Networks can be partially observed: account for missing nodes \refer{CPW12,RAR18}
  \end{itemize}
  
  \pause\bigskip\bigskip
  \paragraph{Application to ecology.}
  \begin{itemize}
   \item Importance of covariates
   \item No guaranty for VEM inference in the PLN network
   \item Need for (reasonably generic) graphical models for count data
  \end{itemize}
