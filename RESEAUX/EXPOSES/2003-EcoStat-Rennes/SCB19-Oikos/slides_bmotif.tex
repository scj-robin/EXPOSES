\frame{
\frametitle{
Interactions indirectes et paradigme des indices (1)}
\begin{block}{Interaction indirecte}
 Impact d'une espèce sur l'autre via des espèces intermédiaires
\end{block}

 \begin{itemize}
\item \textcolor{purple}{Chaîne d'interaction} : le changement d'abondance d'une espèce donatrice peut influencer celle d'une receveuse en affectant celle d'une intermédiaire interagissant avec les deux
\item \textcolor{purple}{Interaction de modification} : un prédateur pourrait ne pas se nourrir dans un patch à cause de la présence d'une plante défendue et donc réduit la consommation d'une autre espèce qui elle est consommable
\end{itemize} Ici focus sur les chaînes d'interactions capturées par les motifs.

\frame{
\frametitle{
Interactions indirectes - réseau plante-pollinisateur}
\begin{itemize}
\item Communauté plante-pollinisateur aux Seychelles
\item Retrait de plantes exotiques $\Rightarrow$ espèces pollinisatrices plus généralistes
\item 2 espèces abondantes et très généralistes : \emph{Lasioglossum mahense} (native) et \emph{Apis mellifera} (non native)
\item Ont-elles les mêmes stratégies de sélection et rôles dans la communauté ?
\begin{itemize}
\item via la métrique $d^{\prime}$ à l'échelle de l'espèce  : ces espèces sont des super généralistes
\item via la signature des positions dans les motifs : \emph{Lasioglossum mahense} en grande compétition avec les autres généralistes et \emph{Apis mellifera} visite les plantes spécialistes et en compétition avec les autres pollinisateurs spécialistes
\end{itemize}
\end{itemize}
}

\frame{
\frametitle{
Applications potentielles -- Limitations et défis}
\begin{block}{Exemples de questions auxquelles on pourrait répondre via l'étude des motifs}
\begin{itemize}
\item \'Etude à l'échelle des espèces
\item Focus sur le changement de structure dans le temps et l'espace
\item Performances prédictives 
\item Compréhension des conséquences fonctionnelles de la structure des communautés
\end{itemize}
\end{block}

\begin{block}{Limitations et défis}
\begin{itemize}
\item Motifs uniquement à partir de réseaux non valués 
\item Relation non bijective entre signatures à partir du motif et le réseau
\end{itemize}
\end{block}
}



\frame{

Conclusion}
\begin{block}{Leur conclusion} 
\begin{itemize}
\item \'Etude des relations indirectes essentielle pour comprendre le rôle des espèces et la structure des interactions biotiques
\item L'utilisation des motifs doit devenir aussi courante que celle des indices usuels dans l'analyse des réseaux bipartites
\end{itemize}
\end{block}
