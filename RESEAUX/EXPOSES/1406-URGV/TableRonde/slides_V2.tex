\bigskip

\section{Quel représentation ?}
%====================================================================
\frame{\frametitle{Graphe (au sens math\'ematique)}

\begin{tabular}{cc}
  \begin{tabular}{p{.4\textwidth}}
  Un graphe $G$ est constitu\'e 
  \begin{itemize}
   \item de noeuds $(V)$  
   \item et d'arêtes $(E)$: 
  \end{itemize}
  $$
  G = (V, E)
  $$
  \end{tabular} 
  &
  \begin{tabular}{p{.5\textwidth}}
  \includegraphics[height=.5\textheight]{graphe.jpeg}
  \end{tabular} 
\end{tabular}

  \begin{itemize}
  \item Graphe orienté: $\{i \rightarrow j\} \neq \{j \rightarrow i\}$ / non-orienté: $\{i \sim j\}$
  \item Composante connexe: ensemble de noeuds connectés entre eux par au moins un chemin
  \end{itemize}
  
  \bigskip
  \paragraph{$\longrightarrow$} Description (statique) des interactions existantes

}

%====================================================================
\frame{\frametitle{Modèle graphique}


Représentation probabiliste des dépendances entre des variables (i.e. expression de gènes).

\begin{tabular}{cc}
  \begin{tabular}{p{.5\textwidth}}
  $$
  \begin{array}{l}
  P(A, B, C, D, E) \\
  ~ \\
  \quad = P(A) \; P(B|A) \; P(C|A) \\
  ~ \\
  \qquad \times \; P(D) \; P(E | C, D)
  \end{array}
  $$
  \end{tabular} 
  &
  \begin{tabular}{p{.5\textwidth}}
  \includegraphics[height=.4\textheight]{Fig-GM-SimpleExample.png}
  \end{tabular} 
\end{tabular}

\bigskip
\begin{itemize}
 \item Autres noms: ``réseaux bayésiens'', ``DAG'', ...
 \item Structure: peut-être donnée ou à inférer
\end{itemize}

}
