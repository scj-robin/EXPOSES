\frame{ \frametitle{A null model}

  \newcommand{\obs}{\text{obs}}

  \paragraph{Question:} is motif $m$ unexpectedly frequent (or rare) in a given observed network?
  $$
  N_m(Y_{\obs}) = \text{ number of occurrence of $m$ in the network $Y_{\obs}$}
  $$
  \ra \url{bmotif} R package 
  
  \bigskip \bigskip \pause 
  \paragraph{Significance:} define a null model to evaluate
  $$
  \text{p-value } = P_0(N_m(Y) \geq N_m(Y_{\obs}))
  $$

  \bigskip \pause 
  \paragraph{A null model:} bipartite configuration model (keeping top and bottom degrees fixed)
  \begin{align*}
    \{Y^b\}_{b = 1 \dots B} \text{ iid } & \sim \text{ bipartite configuration model} \\
    N_m(Y^b) & = \text{ number of occurrence of $m$ in $Y^b$} \\
    \text{p-value } & = \frac1B \# \{b: N_m(Y^b) \geq N_m(Y_{\obs})\}
  \end{align*}
  \ra Need to \url{bmotif} $B$ times
}
