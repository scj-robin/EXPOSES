\frame{\frametitle{Barents sea fish abundances: Model selection}

  \paragraph{BIC for composite likelihood.} \refer{GaS10}
  $$
  BIC = \cl_{\theta}(Y) - \frac{\log(n)}2 \text{dim}(\theta) 
  \qquad \qquad 
  \text{with} \quad
  \text{dim}(\theta) = \tr[J(\theta) H(\theta)^{-1}].
  $$
