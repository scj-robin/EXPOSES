\frame{\frametitle{Some differences} 

  \paragraph{Nature of the latent variables:}
  \begin{description}
   \item[PLN] continuous, multivariate: $Z_i \sim \Ncal$
   \item[SBM] univariate, discrete: $Z_i \sim \Mcal$
  \end{description}
  
  \bigskip \bigskip \pause
  \paragraph{Modeling function of the latent variables:}
  \begin{description}
   \item[PLN] \bigskip instrumental: comfortable multivariate Gaussian setting \\
   \ra no precise ecological interpretation of the $Z_i$ \\
   \ra focus on the inference of the parameter $\theta = (\beta, \Sigma)$
   \item[SBM] \pause \bigskip 'mechanistic': hopefully, species groups have an ecological meaning \\
   \ra easier interpretation of the $Z_i$ \\
   \ra focus on unsupervised classification
  \end{description}
  
  \bigskip \bigskip \pause
  \paragraph{Emission distribution:} mostly Poisson in these lectures (a lot of count data in ecology) \\ ~\\ 
  \ra Poisson log-normal and block-models are amendable to any (parametric) distribution

}
